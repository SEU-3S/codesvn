<script>
var mCal;
 
window.onload = function() {
    mCal = new dhtmlxCalendarObject('dhtmlxCalendar', true);
}
function hide_show() {
    var bt = document.getElementById('bt');
    if (mCal.isVisible()); {
        bt.value = 'SHOW';
        mCal.hide();
    } else; {
        mCal.show();
        bt.value = 'HIDE';
    }
}
</script>
