window.onload = function() {
 
    mCal = new dhtmlxCalendarObject('dhtmlxCalendar', false, {
        isYearEditable: true
    });
    mCal.loadUserLanguage('en-us');
    mCal.setYearsRange(2000, 2500);
    mCal.draw();
 
}
function chLang(lang) {
    mCal.loadUserLanguage(lang);
