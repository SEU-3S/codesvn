var dhxList,
taData,
j = 0;
function doOnLoad() {
    taData = document.getElementById("taData");
    dhxList = new dhtmlXForm("listObj");
    dhxList.attachEvent("onXLS", function() {
        writeLog("<b>onXLS</b> event fired.");
    });
    dhxList.attachEvent("onXLE", function() {
        writeLog("<b>onXLE</b> event fired.");
    });
    dhxList.loadStruct("../common/dhxform.xml?etc=" + new Date().getTime(), function() {
        writeLog("<b>onLoadFunction</b> called.");
    });
}
function writeLog(logText) {
    taData.innerHTML = (++j) + ". " + logText + "<br>" + taData.innerHTML;
}
