var dhxList;
function doOnLoad() {
    dhxList = new dhtmlXForm("listObj");
    dhxList.loadStruct("../common/dhxform2.xml?" + new Date().getTime(), function() {
        dhxList.checkItem("custom_colors");
    });
}
function changeCase(howto) {
    var ids = ["font_color", "background_color"];
    for (var w = 0; w < ids.length; w++) {
        var opts = dhxList.getOptions(ids[w]);
        for (var q = 0; q < opts.length; q++)
            opts[q].text = String(opts[q].text)[howto]();
    }
}
</script>
